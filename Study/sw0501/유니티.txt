

장애물 이동
변수에 transform.position.x 를 대입하고 
new Vector3(x,y,z) 를 이용하여 위치를 변경시킬수 있다
물체의 위치에따라 속도조절,방향변경을 할 경우
변수를 하나 더 선언하여 기준으로 잡고 가능

	[x축을 움직이는 장애물]
	float delta = 0.1f
	float newXPosition = transform.position.x - delta;
	transform.position.x = new Vector3(newXPosition, y , z);
	if (transform.position.x > 1.0)(delta = 0.1f);
	else if (transform.position.x < -1.0)(delta = -0.1f);

사용자 입력 받기
	[입력을 받아 물체를 회전]
	float zRotation = transform.localEulerAngles.z;
	zRotation = zRotation - Input.GetAxis("Horizontal");
	transform.localEulerAngles = new Vector3(0, 0, zRotation);

	if (Input.GetKeyDown (KeyCode.Space)){ 
	GetComponent<Rigidbody>().AddForce(Vector3.up * 500) ; }
  	else if (Input.GetKeyUp(KeyCode.Space)){
            GetComponent<Rigidbody>().AddForce(Vector3.up * 0) ; }
